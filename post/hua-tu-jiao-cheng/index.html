A¶

In [8]:
# 用rcParams，您可以自定义图像，例如 字体font.family，大小，线条颜色，线条宽度等
# 它们是自动运行和启动的

# 以下设置将为所有其他图设置figsize
plt.rcParams['figure.figsize'] = [6, 6] # 宽x高（英寸）
plt.rcParams['figure.figsize'] = 6, 6  # 同上

# 更新字体大小
plt.rcParams.update({'font.size': 26})


# 用（x，y）坐标绘制一个图形。
plt.plot(data['height'], data['mass'], marker='+',  color='black' , linestyle='')
plt.show() # 显示一个图。下一个绘图将创建新的图形。如果没有plt.show()，它将添加到之前的绘图中

# 再一次修改字体大小
plt.rcParams.update({'font.size': 14})

# 用线将点相连
plt.plot(data['height'], data['mass'], marker='+',  color='black' , linestyle='--')
plt.show() 


# 增加标签
plt.plot(data['height'], data['mass'], 'k+') # k +是color ='black'，marker ='+'，linestyle =''的缩写，请看https://matplotlib.org/api/_as_gen/matplotlib.pyplot.plot.html
plt.title('plot 3')
plt.ylabel('y label')
plt.xlabel('x label')

plt.show()

添加点，线和图例¶

In [14]:
plt.rcParams['figure.figsize'] = [6, 6]
plt.plot(data['height'],data['mass'], 'ro', label='data') 

# 添加x和y坐标的点
plt.plot(4,10, color='black', marker='+', markersize=15, label='outlier')
plt.plot(*[4,11], color='black', marker='+', markersize=15, label='unknown')

# 添加一条线
# [[x1,x2], [y1,y2]]
plt.plot([8,0], [6,0] ,'g--')
plt.plot((8,0), (6,0) ,'g--') # Both work

# [[x,y], [x,y]]
# zip([1, 2], [10,11]) -> [(1, 10), (2, 11)]
# *[(1, 10), (2, 11)] -> (1, 10) (2, 11) ; unpacking the containers
plt.plot(*list(zip([1, 2], [10,11])) ,'k-')

# 添加带有x和y坐标的线
plt.legend(markerscale=1, frameon=True, loc='upper left')
plt.show()
